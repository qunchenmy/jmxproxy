001package com.github.mk23.jmxproxy.conf;
002
003import com.fasterxml.jackson.core.JsonParser;
004import com.fasterxml.jackson.core.JsonProcessingException;
005import com.fasterxml.jackson.core.JsonToken;
006import com.fasterxml.jackson.databind.DeserializationContext;
007import com.fasterxml.jackson.databind.JsonDeserializer;
008
009import io.dropwizard.util.Duration;
010
011import java.io.IOException;
012
013/**
014 * <p>Custom deserializer for configuration Duration fields.</p>
015 *
016 * Converts a String or Long into <code>io.dropwizard.util.Duration</code> types to milliseconds.
017 *
018 * @see <a href="http://dropwizard.github.io/dropwizard/0.9.3/dropwizard-util/apidocs/io/dropwizard/util/Duration.html">io.dropwizard.util.Duration</a>
019 * @see <a href="https://fasterxml.github.io/jackson-databind/javadoc/2.6/com/fasterxml/jackson/databind/JsonDeserializer.html">com.fasterxml.jackson.databind.JsonDeserializer</a>
020 *
021 * @since   2016-07-02
022 * @author  mk23
023 * @version 3.3.3
024 */
025public class DurationDeserializer extends JsonDeserializer<Duration> {
026    /** {@inheritDoc} */
027    @Override
028    public final Duration deserialize(
029        final JsonParser jp,
030        final DeserializationContext ctxt
031    ) throws IOException, JsonProcessingException {
032        return jp.getCurrentToken() == JsonToken.VALUE_NUMBER_INT
033            ? Duration.milliseconds(jp.getValueAsLong())
034            : Duration.parse(jp.getValueAsString());
035    }
036}
